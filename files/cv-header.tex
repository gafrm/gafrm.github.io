  \href{#1}{%
    \makebox[1.45em][c]{\textcolor{black}{#2}}%

  }%
}

% Degree and position entries with dates aligned consistently on the right.
\newcommand{\cventry}[4]{%
  \Needspace{4\baselineskip}%
  \begin{tabularx}{\textwidth}{@{}L R{0.23\textwidth}@{}}
    \textbf{#1} & \textcolor{cvmuted}{#2}\\[-0.05em]
    \textcolor{cvmuted}{#3} &
  \end{tabularx}%
  \ifstrempty{#4}{}{%
    \par\vspace{-0.08em}{\small\textcolor{cvmuted}{#4}}%
  }%
  \par\vspace{0.48em}%
}
